                NOT = { is_core_of = ROOT }
                NOT = { is_claimed_by = ROOT }
            }
            AFG = { transfer_state = PREV }
        }
        puppet = AFG
    }
    option = { # Annex
        name = annexation.occupation
        ai_chance = { factor = 5 }
        set_country_flag = AFG_OCCUPIED

    }
}
