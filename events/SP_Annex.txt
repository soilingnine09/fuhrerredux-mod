    id = annex.1

    desc = annex.1.d

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            BRM = {
                add_state_core = 288
                add_state_core = 640
                add_state_core = 812
                add_state_core = 805
                add_state_core = 1007
            }
        }
    }

    option = {
        name = annex.1.a

        release = BRM
        if = {
            limit = { has_dlc = "Together for Victory" }
            set_autonomy = {
                autonomy_state = SP_sponsored_state
                target = BRM
            }
        }
        if = {
            limit = {
                NOT = { has_dlc = "Together for Victory" }
            }
            puppet = BRM
        }
    }

    option = {
        name = annex.1.b

        release = BRM
        add_political_power = 50
    }
}
